
uniform mat3 uBackgroundM;
uniform mat3 uForegroundM;

attribute vec2 aPosition;

varying vec2 vTexPosBackground;
varying vec2 vTexPosForeground;

void main() {
	gl_Position = vec4(aPosition, 0.0, 1.0);
	
	vTexPosBackground = (transpose(uBackgroundM) * vec3(aPosition, 1.0)).xy;
	vTexPosForeground = (transpose(uForegroundM) * vec3(aPosition, 1.0)).xy;
}
